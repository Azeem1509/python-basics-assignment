# Task 1: Variables and Data Types
name = "Azeem"
age = 31
height = 5.50
is_student = True

# Printing all four values in a single statement
print(name, age, height, is_student)

# Task 2: String Formatting
# We use an f-string (formatted string literal) to embed variables directly
print(f"Name: {name} | Age: {age} | Height: {height} | Student: {is_student}")

# Task 3: Arithmetic Operations
age_in_months = age * 12
age_in_days = age * 365
remainder_by_7 = age % 7
age_squared = age ** 2

print(f"Age in months: {age_in_months}")
print(f"Age in days: {age_in_days}")
print(f"Remainder when divided by 7: {remainder_by_7}")
print(f"Age raised to the power of 2: {age_squared}")
